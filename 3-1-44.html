def product(A, B):
    """
    >>a = [[1,2, 3], [4, 5, 6]]
    >>b = [[1, 1], [1, 1], [1, 1]]
    >>print product(a,b) == [[6,6,],[15,15]]
    """
    AH = len(A)
    AW = len(A[0])
    BH = len(B)
    BW = len(B[0])
    if AW != BH:
        print "Size Error"
        print AW, BH
    res = [[0 for x in range(BW)] for y in range(AH)]
    for i in range(AH):
        for j in range(BW):
            for k in range(AW):
                res[i][j] += A[i][k] * B[k][j]
    return res

def transpose(A):
    """
    >>>a = [[1,2], [3,4]]
    >>>transpose(a) == [[1,3],[2,4]]
    True
    """
    AH = len(A)
    AW = len(A[0])
    res = [[0 for x in range(AH)] for y in range(AW)]
    for i in range(AH):
        for j in range(AW):
            res[j][i] = A[i][j]
    return res

def plus(A, B):
    """
    >>>a = [[1,2], [3,4]]
    >>>b = [[1, 0], [0, 1]]
    >>>plus(a,b) == [[2,2],[3,5]]
    True
    """
    AH = len(A)
    AW = len(A[0])
    res = [[0 for x in range(AW)] for y in range(AH)]
    for i in range(AH):
        for j in range(AW):
            res[i][j] = A[i][j] + B[i][j]
    return res

def minus(A, B):
    """
    >>>a = [[1,2], [3,4]]
    >>>b = [[1, 0], [0, 1]]
    >>>minus(a,b) == [[0,2],[3,3]]
    True
    """
    AH = len(A)
    AW = len(A[0])
    res = [[0 for x in range(AW)] for y in range(AH)]
    for i in range(AH):
        for j in range(AW):
            res[i][j] = A[i][j] - B[i][j]
    return res

def times(num, A):
    """
    >>>a = [[1,2], [3,4]]
    >>>times(2,a) == [[2,4],[6,8]]
    True
    """
    AH = len(A)
    AW = len(A[0])
    for i in range(AH):
        for j in range(AW):
            A[i][j] = A[i][j] * num
    return A
